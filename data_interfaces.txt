RegisterContact
    input -- ComData{ command=RegisterContact, id=null, data=ContactName }
    output -- ResData{ command=ContactId, data=ContactId }

Disconnect:
	input -- ComData{ command=Disconnect, id=null, data=null }
	output -- null

Refresh contacts:
    input -- ComData{ command=RefreshContacts, id=null, data=null }
    output -- ResData{ resultType=UpdatedContacts, data=Contacts }

Create conference:
    input -- ComData{ command=CreateConference, id=null, data=Conference{ 
            name=ConferenceName;
            id=null;
            members=ArrayList< ContactId >;
        }
    }
    output -- 1. ResData{ resultType=CreatedConference, data=ConferenceId }
              2. ResData{ resultType=AddedToNewConference, data=Conference }

Add to conference:
    input -- ComData{ command=AddToConference, id=null, data=ContactConfPair }
    output -- 
        1. ResData{ resultType=AddedToConferece, data=ContactConfPair }
        2. ResData{ resultType=AddedConference, data=Conference (full with me) }

Remove from conference:
    input -- ComData{ command=RemoveFromConference, id=null, data=ContactConfPair }
    output -- ResData{ resultType=RemovedFromConference, data=ContactConfPair }

Delete conference:
    input -- ComData{ command=DeleteConference, id=ConferenceId, data=null }
    output -- ResData{ resultType=DeletedConference, data=ConferenceId }

Send message to contact:
    input -- ComData{ command=SendMessageToContact, id=ContactId; data=Message }
    output -- ResData{ command=Message, data=ContactMessagePair }

Send message to conference: (Проверка на отправку в конференцию, в которой не состоишь? - Не надо!)
    input -- ComData{ command=SendMessageToConference, id=ConferenceId, data=Message }
    output -- Broadcast -> ResData={ resultType=Message, data=ContactMessagePair }

Refresh files:
    input -- ComData{ command=RefreshStorage, id=null, data=null }
    output -- ResData{ resultType=UpdatedFiles, data=ArrayList< File > }

Upload file to the storage:
    input -- ComData{ command=UploadFile, id=null, data=File{
            name=FileName;
            data=array of bytes of the file;
        }
    }
    output -- ResData{ resultType=UploadedFile, data=bool }

Download File from the Storage:
    input -- ComData{ command=DownloadFile, id=FileId, data=null }
    output -- ResData{ resultType=DownloadedFile, data=FileData }

Remove file from the storage:
    input -- ComData{ command=RemoveFile, id=FileId, data=null}
    output -- ResData{ resultType=RemovedFile, data=bool }
